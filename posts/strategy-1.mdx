export async function* tour(ct, opts = ct.api.parseArgsAsJs(ct.args, { to: 'array' })) {
  for (const to of opts.to) {
    yield* ct.lib.move(ct, { npcKey: opts.npcKey, to });
    await ct.api.sleep(opts.pauseMs ?? 0.8);
  }
}
